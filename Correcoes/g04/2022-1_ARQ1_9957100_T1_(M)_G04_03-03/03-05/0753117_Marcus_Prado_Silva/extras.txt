06.) Dada a express찾o em Verilog abaixo, identificar as express천es SoP equivalente:

module FXYZ (output S1, input X, input Y, input Z);
   assign s1 = ( X | ~Y | ~Z ) & (~X | Y | Z);
endmodule // FXYZ


----------------------------------------------------
Resposta:

   PoS = (X + ~Y + ~Z) . (~X + Y + Z) 
         /// 100 = 4      /// 011 = 3

07.)Dada a express찾o em Verilog abaixo, identificar as express천es PoS equivalente:


module fxyz (output s1, input x, input y, input z);
   assign s1 = ( x & ~y & ~z ) | (~x & y & z);
endmodule // fxyz

----------------------------------------------------

   SoP = ( x & ~y & ~z ) | (~x & y & z)
